Edit a row in your spreadsheet
Javascript Example
  
    let url = 'http://localhost:5000/api/sheets/{{ sheet.google_spreadsheet_id }}/2';
    let body = {
        ...
    }
    fetch(url, {
        method: 'PUT',
        body: JSON.stringify(body)
    })
        .then((response) => response.json())
        .then(json => {
            // Do something with object
            console.log(json.sheet1);
        });
