GET
Retrieve rows from your spreadsheet
Javascript Example
let url = 'http://localhost:5000/api/sheets/{{ sheet.google_spreadsheet_id }}';
fetch(url)
    .then((response) => response.json())
    .then(json => {
        // Do something with the data
        console.log(json.sheet1S);
    });
POST
Add a row to your spreadsheet
Javascript Example
let url = 'http://localhost:5000/api/sheets/{{ sheet.google_spreadsheet_id }}';
let body = {
  ...
}
fetch(url, {
  method: 'POST',
  body: JSON.stringify(body)
})
.then((response) => response.json())
.then(json => {
  // Do something with object
  console.log(json.sheet1);
});

  ...
}

  method: 'PUT',
  body: JSON.stringify(body)
})
.then((response) => response.json())
.then(json => {
  // Do something with object
  console.log(json.sheet1);
});

Delete a row in your spreadsheet
Javascript Example
let url = 'http://localhost:5000/api/sheets/{{ sheet.google_spreadsheet_id }}/2';
fetch(url, {
  method: 'DELETE',

  console.log('Object deleted');
